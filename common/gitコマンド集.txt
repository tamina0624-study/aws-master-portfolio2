■git コマンド

# 1. 変更があったことを確認
git status

# 2. すべての変更をステージング
git add .

# 3. 修正内容をメモ（コミット）
git commit -m "機密情報のマスクツール-react版追加"

# 4. GitHubに送信
git push origin main

#commitを取り消す
git reset --soft HEAD~1

#git index 修正
git rm --cached -r .

#git logの削除
 git log --all -- path/to/file

 #キャッシュ削除
 Remove-Item -Recurse -Force $HOME\.cache\pre-commit

■gitからファイルを消したい場合
git clone --mirror https://github.com/ユーザー名/リポジトリ.git
cd リポジトリ.git

git filter-repo --path <削除したいファイルのgitpath> --invert-paths
git push --force --all
git push --force --tags
